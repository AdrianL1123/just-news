# JustNews Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a minimal, text-only news reader that fetches from TheNewsAPI via a Hono/Cloudflare proxy, with an optional Ollama-powered daily briefing per category.

**Architecture:** Vite + React SPA served via Cloudflare Pages. A Cloudflare Pages Function (`functions/api/[[route]].ts`) acts as a thin Hono proxy that injects the API key server-side. The frontend uses a simple in-memory cache (5-min TTL) via a `useNews` hook. The Ollama summary feature calls `localhost:11434` directly from the browser — personal tool, local only.

**Tech Stack:** Vite 5, React 18, TypeScript 5 (strict), Tailwind CSS v3, shadcn/ui, Lucide React, Hono, Wrangler (Cloudflare Pages), pnpm, DM Serif Display + IBM Plex Sans fonts via fontsource

```typescript
export function relativeTime(dateStr: string): string {
  const seconds = Math.floor((Date.now() - new Date(dateStr).getTime()) / 1000)
  if (seconds < 60) return 'just now'
  if (seconds < 3600) return `${Math.floor(seconds / 60)}m ago`
  if (seconds < 86400) return `${Math.floor(seconds / 3600)}h ago`
  return `${Math.floor(seconds / 86400)}d ago`
}

export function formatUpdatedAt(date: Date): string {
  const seconds = Math.floor((Date.now() - date.getTime()) / 1000)
  if (seconds < 60) return 'just now'
  if (seconds < 3600) return `${Math.floor(seconds / 60)} min ago`
  return `${Math.floor(seconds / 3600)}h ago`
}
```

```typescript
import { Hono } from 'hono'
import { handle } from 'hono/cloudflare-pages'
import { cors } from 'hono/cors'

type Bindings = {
  NEWS_API_KEY: string
}

const app = new Hono<{ Bindings: Bindings }>()

app.use('*', cors())

app.get('/api/news', async (c) => {
  const locale = c.req.query('locale') ?? 'us'
  const category = c.req.query('category') ?? 'general'
  const apiKey = c.env.NEWS_API_KEY

  if (!apiKey) {
    return c.json({ error: 'API key not configured' }, 500)
  }

  const url = new URL('https://api.thenewsapi.com/v1/news/top')
  url.searchParams.set('api_token', apiKey)
  url.searchParams.set('locale', locale)
  url.searchParams.set('language', 'en')
  url.searchParams.set('limit', '3')

  if (category !== 'general') {
    url.searchParams.set('categories', category)
  }

  const res = await fetch(url.toString())
  if (!res.ok) {
    return c.json({ error: `Upstream error: ${res.status}` }, 502)
  }

  const data = await res.json()
  return c.json(data)
})

export const onRequest = handle(app)
```

- [ ] **Step 1: Create src/hooks/useNews.ts**

```typescript
import { useState, useEffect, useCallback, useRef } from 'react'
import type { Article } from '@/types'

const CACHE_TTL = 5 * 60 * 1000

interface CacheEntry {
  articles: Article[]
  timestamp: number
}

const cache = new Map<string, CacheEntry>()

interface UseNewsResult {
  articles: Article[]
  loading: boolean
  refreshing: boolean
  error: string | null
  updatedAt: Date | null
  refresh: () => void
}

export function useNews(locale: string, category: string): UseNewsResult {
  const [articles, setArticles] = useState<Article[]>([])
  const [loading, setLoading] = useState(false)
  const [refreshing, setRefreshing] = useState(false)
  const [error, setError] = useState<string | null>(null)
  const [updatedAt, setUpdatedAt] = useState<Date | null>(null)
  const [tick, setTick] = useState(0)
  const isBust = useRef(false)

  useEffect(() => {
    const key = `${locale}:${category}`
    const bust = isBust.current
    isBust.current = false

    if (!bust) {
      const hit = cache.get(key)
      if (hit && Date.now() - hit.timestamp < CACHE_TTL) {
        setArticles(hit.articles)
        setUpdatedAt(new Date(hit.timestamp))
        return
      }
    }

    bust ? setRefreshing(true) : setLoading(true)
    setError(null)

    let cancelled = false

    fetch(
      `/api/news?locale=${encodeURIComponent(locale)}&category=${encodeURIComponent(category)}`
    )
      .then((r) => {
        if (!r.ok) throw new Error(String(r.status))
        return r.json() as Promise<{ data?: Article[] }>
      })
      .then((json) => {
        if (cancelled) return
        const data = json.data ?? []
        cache.set(key, { articles: data, timestamp: Date.now() })
        setArticles(data)
        setUpdatedAt(new Date())
      })
      .catch(() => {
        if (!cancelled) setError('Could not load news. Try refreshing.')
      })
      .finally(() => {
        if (!cancelled) {
          setLoading(false)
          setRefreshing(false)
        }
      })

    return () => {
      cancelled = true
    }
  }, [locale, category, tick])

  const refresh = useCallback(() => {
    isBust.current = true
    setTick((t) => t + 1)
  }, [])

  return { articles, loading, refreshing, error, updatedAt, refresh }
}
```

### Task 11: useSummary Hook

**Files:**
- Create: `src/hooks/useSummary.ts`

- [ ] **Step 1: Create src/hooks/useSummary.ts**

```typescript
import { useState, useCallback } from 'react'
import type { Article } from '@/types'

interface UseSummaryResult {
  summary: string | null
  loading: boolean
  error: string | null
  generate: (articles: Article[], categoryLabel: string) => Promise<void>
  clear: () => void
}

export function useSummary(): UseSummaryResult {
  const [summary, setSummary] = useState<string | null>(null)
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const generate = useCallback(
    async (articles: Article[], categoryLabel: string) => {
      setLoading(true)
      setSummary(null)
      setError(null)

      const headlines = articles.map((a) => `- ${a.title}`).join('\n')
      const prompt = `Summarize these ${categoryLabel} news headlines in 2-3 sentences. Be concise and factual. Do not use bullet points or headers.\n\n${headlines}`

      try {
        const res = await fetch('http://localhost:11434/api/generate', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            model: 'llama3.2',
            prompt,
            stream: false,
          }),
        })
        if (!res.ok) throw new Error('ollama_http_error')
        const data = (await res.json()) as { response: string }
        setSummary(data.response.trim())
      } catch {
        setError(
          'Ollama is not running. Install it at ollama.com then run: ollama pull llama3.2 && ollama serve'
        )
      } finally {
        setLoading(false)
      }
    },
    []
  )

  const clear = useCallback(() => {
    setSummary(null)
    setError(null)
  }, [])

  return { summary, loading, error, generate, clear }
}
```

```typescript
import { useState, useEffect } from 'react'
import { Sparkles } from 'lucide-react'
import { cn } from '@/lib/utils'
import type { Article, NavState } from '@/types'
import { ArticleRow } from './ArticleRow'
import { SummaryCard } from './SummaryCard'
import { useSummary } from '@/hooks/useSummary'

interface Props {
  articles: Article[]
  loading: boolean
  error: string | null
  nav: NavState
}

export function Feed({ articles, loading, error, nav }: Props) {
  const { summary, loading: summaryLoading, error: summaryError, generate, clear } =
    useSummary()
  const [summaryOpen, setSummaryOpen] = useState(false)

  // Reset summary when category or country changes
  useEffect(() => {
    setSummaryOpen(false)
    clear()
  }, [nav.country.code, nav.category.id, clear])

  const handleSummarize = () => {
    if (summaryOpen) {
      setSummaryOpen(false)
      clear()
      return
    }
    setSummaryOpen(true)
    void generate(articles, nav.category.label)
  }

  const canSummarize = !loading && articles.length > 0

  return (
    <main className="flex-1 min-w-0">
      {/* Summary bar */}
      <div className="px-8 py-3 border-b border-border flex items-center">
        <button
          onClick={handleSummarize}
          disabled={!canSummarize}
          className={cn(
            'flex items-center gap-1.5 px-3 py-1 border border-border text-[12px] text-muted-foreground transition-all',
            'hover:border-primary hover:text-primary hover:bg-primary/[0.08]',
            summaryOpen && 'border-primary text-primary bg-primary/[0.08]',
            !canSummarize && 'opacity-30 pointer-events-none'
          )}
        >
          <Sparkles className="w-3 h-3" />
          {summaryOpen ? 'Hide Summary' : 'Summarize Today'}
        </button>
      </div>

      {/* Summary card */}
      {summaryOpen && (
        <SummaryCard
          summary={summary}
          loading={summaryLoading}
          error={summaryError}
          categoryLabel={nav.category.label}
          onDismiss={() => {
            setSummaryOpen(false)
            clear()
          }}
        />
      )}

      {/* Article list */}
      {loading && (
        <div className="px-8 py-12 text-sm text-muted-foreground/50">
          Loading…
        </div>
      )}

      {!loading && error && (
        <div className="px-8 py-12 text-sm text-muted-foreground/50">{error}</div>
      )}

      {!loading && !error && articles.length === 0 && (
        <div className="px-8 py-12 text-sm text-muted-foreground/50">

        </div>
      )}

      {!loading &&
        !error &&
        articles.map((article) => (

}
```
